TiGateway has been installed!

1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Values.namespace | default "default" }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "tigateway.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Values.namespace | default "default" }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status by running 'kubectl get --namespace {{ .Values.namespace | default "default" }} svc -w {{ include "tigateway.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Values.namespace | default "default" }} {{ include "tigateway.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.ports.gateway }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Values.namespace | default "default" }} -l "{{ include "tigateway.selectorLabels" . | replace "=" "=" | replace ":" ":" }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Values.namespace | default "default" }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Values.namespace | default "default" }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Access the Admin UI:
   Admin UI is available on port {{ .Values.service.ports.admin }}
   Management endpoints are available on port {{ .Values.service.ports.management }}

3. Check the status of your deployment:
   kubectl get pods --namespace {{ .Values.namespace | default "default" }} -l "{{ include "tigateway.selectorLabels" . | replace "=" "=" | replace ":" ":" }}"

4. View logs:
   kubectl logs --namespace {{ .Values.namespace | default "default" }} -l "{{ include "tigateway.selectorLabels" . | replace "=" "=" | replace ":" ":" }}" -f

For more information, visit: https://github.com/tigateway/tigateway
